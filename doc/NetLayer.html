def initWeights(size_y, size_x = size_y, min = -1.0, max = 1.0)
	r = Random.new
	w = Matrix.new(size_y, size_x)
	(0...size_y).each do |y|
		(0...size_x).each do |x|
			w[y, x] = r.rand(min..max)
		end
	end
	return w
end
