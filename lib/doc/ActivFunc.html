def reLu(m)
	Matrix.setVectorizedMatrix(
	      m.matrix.map do |x|
	      	x > 0 ? x : 0
		end,
		m.size_y,
		m.size_x
	)
end

def sigmoidPrime(m)
	Matrix.setVectorizedMatrix(
	      m.matrix.map do |x|
	      	CMath.exp(-x) / (1.0 + CMath.exp(-x))**2
		end,
		m.size_y,
		m.size_x
	)
end

def softMax(m)
	Matrix.set((0...m.size_y).map do |y|
		sum = 0.0
		out = (0...m.size_x).map do |x|
			tmp = CMath.exp(m[y, x])
			sum += tmp
			tmp
		end
		out.map { |val| (val / sum) }
	end)
end

def tanhPrime(m)
	Matrix.setVectorizedMatrix(
	      m.matrix.map do |x|
			1.0 - CMath.tanh(x)**2
		end,
		m.size_y,
		m.size_x
	)
end
